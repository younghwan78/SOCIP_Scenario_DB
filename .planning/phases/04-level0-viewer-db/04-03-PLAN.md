`_sw_stack_to_view_response(projection: dict) -> ViewResponse` 함수 신규 작성:

```python
def _sw_stack_to_view_response(projection: dict) -> ViewResponse:
    """topology mode: sw_stack 섹션 → SW 노드 + SW→HW control edges."""
    pipeline = projection.get("pipeline", {})
    sw_stack_raw = pipeline.get("sw_stack", [])

    # 레이어별 x좌표 누적 (같은 레이어에 여러 노드가 있을 경우 오른쪽으로 배치)
    from scenario_db.view.layout import LANE_Y, LANE_LABEL_W
    LAYER_X_START = {
        "app":       LANE_LABEL_W + 150,
        "framework": LANE_LABEL_W + 150,
        "hal":       LANE_LABEL_W + 150,
        "kernel":    LANE_LABEL_W + 150,
    }
    X_STEP = 180  # 같은 레이어 내 노드 간 간격

    layer_x_counter: dict[str, int] = {}  # 레이어별 노드 누적 인덱스

    nodes: list[NodeElement] = []
    sw_node_ids: set[str] = set()

    for sw in sw_stack_raw:
        layer = sw.get("layer", "app")
        nid = sw.get("id", "")
        label = sw.get("label", nid)
        idx = layer_x_counter.get(layer, 0)
        x = float(LAYER_X_START[layer] + idx * X_STEP)
        y = LANE_Y[layer]
        layer_x_counter[layer] = idx + 1
        sw_node_ids.add(nid)
        nodes.append(NodeElement(
            data=NodeData(id=nid, label=label, type="sw", layer=layer),
            position={"x": x, "y": y},
        ))

    # HW 노드도 포함 (topology mode에서 SW→HW 연결을 위해)
    # architecture mode의 _projection_to_view_response()에서 생성한 것과 동일하게 생성
    # (단, visible_layers가 ["app","framework","hal","kernel","hw"]이므로 hw 노드도 렌더링)
    hw_view = _projection_to_view_response(projection)
    nodes.extend(hw_view.nodes)

    # SW→HW control edges (ip_ref가 있는 sw_stack 노드에서 HW 노드로)
    edges: list[EdgeElement] = []
    # HW 노드 id 목록 (pipeline.nodes에서 추출)
    hw_node_ids = {n["id"] for n in pipeline.get("nodes", [])}

    for sw in sw_stack_raw:
        nid = sw.get("id", "")
        ip_ref = sw.get("ip_ref")  # pipeline node id (e.g., "csis0")
        if ip_ref and ip_ref in hw_node_ids:
            edge_id = f"e-sw-{nid}-{ip_ref}"
            edges.append(EdgeElement(data=EdgeData(
                id=edge_id,
                source=nid,
                target=ip_ref,
                flow_type="control",   # Literal 제약: "OTF","vOTF","M2M","control","risk"
            )))

    # HW→HW edges도 포함 (아키텍처 구조 보존)
    edges.extend(hw_view.edges)

    summary = ViewSummary(
        scenario_id=projection["scenario_id"],
        variant_id=projection["variant_id"],
        name=projection.get("project_name") or projection["scenario_id"],
        subtitle="topology mode",
        period_ms=0.0,
        budget_ms=0.0,
        resolution="",
        fps=0,
        variant_label="",
    )

    return ViewResponse(
        level=0,
        mode="topology",
        scenario_id=projection["scenario_id"],
        variant_id=projection["variant_id"],
        nodes=nodes,
        edges=edges,
        risks=[],
        summary=summary,
    )
```

`project_level0()` 함수에서 topology 분기 수정:

```python
def project_level0(scenario_id: str, variant_id: str, *, mode: str = "architecture", db: "Session") -> ViewResponse:
    if mode not in ("architecture", "topology"):
        raise NotImplementedError(f"mode '{mode}' is not supported")

    from scenario_db.db.repositories.view_projection import get_view_projection
    from sqlalchemy.exc import NoResultFound
    projection = get_view_projection(db, scenario_id, variant_id)
    if projection is None:
        raise NoResultFound(f"scenario '{scenario_id}' / variant '{variant_id}' not found")

    if mode == "architecture":
        return _projection_to_view_response(projection)
    else:  # topology
        return _sw_stack_to_view_response(projection)
```

Key implementation notes:
- `sw_stack_raw`가 빈 리스트인 경우 SW 노드 없이 HW 노드만 포함한 ViewResponse 반환 (오류 없음)
- `_projection_to_view_response()` 재사용 — hw_view에서 HW 노드와 엣지 추출
- topology mode에서 HW 노드는 architecture mode와 동일한 x좌표 사용 (stage 기반)
- SW 노드 x좌표는 레이어별로 누적 배치 (같은 레이어 내 노드들은 X_STEP 간격으로 우측 배치)
- `flow_type="control"` — Literal 제약 준수 (SW→HW 엣지는 반드시 "control")
- `_sw_stack_to_view_response` 함수는 `_projection_to_view_response` 바로 뒤에 정의

Acceptance criteria:
- [ ] `project_level0(mode="topology")` 호출이 `NotImplementedError`를 발생시키지 않는다
- [ ] topology mode ViewResponse에 SW 노드(layer in {"app","framework","hal","kernel"})가 포함된다
- [ ] SW→HW 엣지의 `flow_type`이 `"control"`이다
- [ ] `uv run pytest tests/integration/test_view_topology.py -x -q` 통과 (Task 4에서 작성)

```python
_GATE_STATUS_STYLE: dict[str, dict[str, str]] = {
    "PASS":            {"border": "#D1D5DB", "bg": "#F9FAFB", "text": "#374151"},
    "WARN":            {"border": "#F59E0B", "bg": "#FFFBEB", "text": "#92400E"},

def render_gate_inspector(gate: GateExecutionResult) -> None:
    """Gate 실행 결과를 인스펙터 패널에 표시 (status badge + matched_rules risk card).

    VIEW-04: GateExecutionResult를 인스펙터 패널 + risk card로 표시.
    """
    status_str = str(gate.status)   # GateResultStatus는 StrEnum → str() 사용
    style = _GATE_STATUS_STYLE.get(status_str, _GATE_STATUS_STYLE["PASS"])

    # ── Status badge ──────────────────────────────────────────────────────
    st.markdown(
        '<p style="font-size:10px;font-weight:700;color:#9CA3AF;'
        'letter-spacing:0.08em;text-transform:uppercase;margin-bottom:6px;">'
        'GATE STATUS</p>',
        unsafe_allow_html=True,
    )
    st.markdown(
        f'<div style="border:2px solid {style["border"]};background:{style["bg"]};'
        f'border-radius:8px;padding:8px 12px;margin-bottom:10px;">'
        f'<span style="font-size:14px;font-weight:700;color:{style["text"]};">'
        f'{status_str}</span></div>',
        unsafe_allow_html=True,
    )

    # ── Matched issues (issue id 목록) ────────────────────────────────────
    if gate.matched_issues:
        st.markdown(
            '<p style="font-size:10px;font-weight:700;color:#9CA3AF;'
            'letter-spacing:0.08em;text-transform:uppercase;margin-bottom:4px;">'
            'MATCHED ISSUES</p>',
            unsafe_allow_html=True,
        )
        issues_html = "".join(
            f'<span style="display:inline-block;background:#FEE2E2;color:#991B1B;'
            f'border-radius:4px;padding:2px 7px;font-size:10px;font-weight:600;'
            f'margin:0 4px 4px 0;">{iss}</span>'
            for iss in gate.matched_issues
        )
        st.markdown(
            f'<div style="margin-bottom:10px;">{issues_html}</div>',
            unsafe_allow_html=True,
        )

    # ── Missing waivers ───────────────────────────────────────────────────
    if gate.missing_waivers:
        st.markdown(
            '<p style="font-size:10px;font-weight:700;color:#9CA3AF;'
            'letter-spacing:0.08em;text-transform:uppercase;margin-bottom:4px;">'
            'MISSING WAIVERS</p>',
            unsafe_allow_html=True,
        )
        waivers_html = "".join(
            f'<span style="display:inline-block;background:#F5F3FF;color:#5B21B6;'
            f'border-radius:4px;padding:2px 7px;font-size:10px;font-weight:600;'

            for w in gate.missing_waivers
        )
        st.markdown(
            f'<div style="margin-bottom:10px;">{waivers_html}</div>',
            unsafe_allow_html=True,
        )

    # ── Matched rules (risk card style) ──────────────────────────────────
    if gate.matched_rules:
        st.markdown(
            f'<p style="font-size:10px;font-weight:700;color:#9CA3AF;'
            f'letter-spacing:0.08em;text-transform:uppercase;margin-bottom:6px;">'
            f'MATCHED RULES ({len(gate.matched_rules)})</p>',
            unsafe_allow_html=True,
        )
        for rule_match in gate.matched_rules:
            rule_status = str(rule_match.result)
            rule_style = _GATE_STATUS_STYLE.get(rule_status, _GATE_STATUS_STYLE["PASS"])
            msg = rule_match.message or ""
            st.markdown(
                f'<div style="background:white;border:1px solid #E5E7EB;border-radius:8px;'
                f'padding:8px 10px;margin-bottom:6px;'
                f'border-left:3px solid {rule_style["border"]};">'
                f'<div style="font-size:11px;font-weight:700;color:#111827;">'
                f'{rule_match.rule_id}</div>'
                f'<div style="font-size:10px;color:#6B7280;margin-top:2px;">{msg}</div>'
                f'</div>',
                unsafe_allow_html=True,
            )
```
