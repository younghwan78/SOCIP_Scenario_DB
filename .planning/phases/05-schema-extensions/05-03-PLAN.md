def test_migration_downgrade_upgrade_cycle(engine, pg):
    """alembic downgrade 0001 → upgrade head 사이클이 오류 없이 완료된다."""
    from alembic import command
    from alembic.config import Config

    # conftest.py의 ALEMBIC_INI 경로 재사용 패턴
    alembic_ini = Path(__file__).parent.parent.parent / "alembic.ini"
    if not alembic_ini.exists():
        pytest.skip("alembic.ini not found")

    url = pg.get_connection_url().replace("postgresql+psycopg2", "postgresql")
    # psycopg2 드라이버로 변환
    if "psycopg2" not in url:
        url = url.replace("postgresql://", "postgresql+psycopg2://")

    cfg = Config(str(alembic_ini))
    cfg.set_main_option("sqlalchemy.url", url)

    # downgrade to 0001
    command.downgrade(cfg, "0001")

    # 신규 컬럼이 사라졌는지 확인
    insp = inspect(engine)
    ip_cols = {c["name"] for c in insp.get_columns("ip_catalog")}
    assert "sim_params" not in ip_cols, "sim_params should be dropped after downgrade"

    # upgrade back to head
    command.upgrade(cfg, "head")

    # 신규 컬럼이 돌아왔는지 확인
    insp2 = inspect(engine)
    ip_cols2 = {c["name"] for c in insp2.get_columns("ip_catalog")}
    assert "sim_params" in ip_cols2, "sim_params should be restored after upgrade"
